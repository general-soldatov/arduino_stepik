```cpp
int led = 1;
void setup() {  
    Serial.begin(115200);  
    pinMode(led, OUTPUT);
}
void loop() {  
    Serial.print("Led ");  
    Serial.print(led);  
    Serial.println(" on");  
    digitalWrite(led, HIGH); 
    delay(1000);  
    Serial.print("Led "); 
    Serial.print(led);  
    Serial.println(" off");  
    digitalWrite(led, LOW);  
    delay(1000);
}
```
